                    dataLayer[key] = dataLayerStaticOverwrites[key];
                }

                if(typeof tmgAdsInitData === 'function') {
                    tmgAdsInitData(tmgAds);

                    tmgAdsInitDFP(tmgAds);
                }
            }

                loadingDiv.innerHTML =  '<div class="loading-content">' +
                                            '<a href="' + window.location + '"><p>' +
                                                'It looks like something is not quite right<br> with your internet connection.<br> Please refresh the page and retry.' +
                                            '</p></a>' +
                                        '</div>';
            }, 10000);

            xmlhttp.onreadystatechange = function() {
                if (xmlhttp.readyState === XMLHttpRequest.DONE ) {

                        var mainContents = document.getElementsByClassName('main-content'),
                            videoPageMainContents;

                        // append new content template
                        var videoPage = createElement('div', {}, xmlhttp.responseText, false);
                        videoPageMainContents = videoPage.getElementsByClassName('main-content');
                        videoPlayerContents = videoPage.getElementsByClassName('videoPlayer');

                        // remove removeMe DIV
                        // simulating a small delay...
                        if(mainContents.length && videoPageMainContents.length && videoPlayerContents.length) {
                            mainContents[0].innerHTML = videoPageMainContents[0].innerHTML;
                            mainContents[0].className += ' videoRenderer';

                            // just be sure everything is actually written to the DOM before we close the loading

                                var event = new CustomEvent('tmg:page-loaded', { bubbles: true, cancelable: true, detail: undefined });
                                document.dispatchEvent(event);
                            }, 0);
                        } else {
                            // remove playlist GET from URL by redirecting to the article
                            window.location.replace(window.location.origin + window.location.pathname);
                        }
                    }
                    // else status code != 200
                    else {
                        // remove playlist GET from URL by redirecting to the article
                        window.location.replace(window.location.origin + window.location.pathname);
                    }
                }
            }


            videoURL = window.location.protocol + '//' + window.location.host

            /*
            Making it work in local as well as on live
            article/?playlist=
            article.html?playlist=
            */

            // does it end in .html?
            if(window.location.pathname.indexOf('.html', window.location.pathname.length - 5) !== -1) {

<div class="articleBodyText section version-2">
<div class="article-body-text component  version-2">
<div class="component-content">
<p><span class="m_first-letter m_first-letter--flagged">G</span>ym owners should warn their customers not to use dangerous <a href="http://www.telegraph.co.uk/sport/football/11877661/Top-footballers-face-suspicion-of-steroid-abuse.html">steroids</a> in a bid to stem the rise of body-enhancing drug abuse fuelled by “selfie” culture, health experts have said.</p>
<p>The National Institute for Health and Care Excellence wants gyms to actively “target people who are taking, or considering taking, image and performance-enhancing drugs”.</p>
<p>The newly published guidance is part of an effort to crack down on the spiralling abuse of <a href="http://www.telegraph.co.uk/men/health/how-dangerous-is-steroid-addiction/">anabolic steroids</a>, which crime analysts believe are taken by more than 60,000 people a year.</p>
</div>
</div>
</div>

<div class="quote section"><div class="quote component" data-tmg-particle="tmg-pull-quote"><div class="component-content"><q class="quote__content">People are a lot more body-conscious than they were a generation ago and society as whole is a lot more aware of appearance</q><span class="quote__author">Jim McVeigh, Liverpool John Moores University</span></div></div></div>
</div>
</div>
</div>
<div class="articleBodyText section">
<div class="article-body-text component  ">
<div class="component-content">
<p><span class="m_first-letter m_first-letter--flagged">S</span>ide effects of the drug, which are illegal to manufacture or supply but not to possess, can include mood disorders, heart disease and severe acne.</p>
<p>NICE experts want public health professionals to integrate themselves into gym-based and online fitness communities in order to push the message about the dangers of the drugs and growth hormones.</p>
<p>Many performance-enhancing substances are injectable and come with the risk of blood-borne disease like HIV, while unscrupulous online dealers often sell counterfeit drugs.</p>
<p>“We want to help prevent people at risk of drug misuse from taking drugs and we also want to prevent people who already use drugs from using them regularly and to excess,” said Professor Gillian leng, deputy chief executive at NICE.</p>
<p>The new guidance also recommends providing danger awareness materials at nightclubs and music festivals.</p>
<p>Jim McVeigh, an expert in public health and one of the academics consulted by NICE, said the most commonly abused drugs were those to enhance muscularity.</p>
<p>“The modern image has changed,” he said.</p>
<p>“People are a lot more body-conscious than they were a generation ago and society as whole is a lot more aware of appearance.”</p>
<p>This, combined with the new “selfie” culture and the ease of acquiring steroids and growth hormones online, was pushing the rise in use, he said.</p>
<p>Other drugs targeted by the new NICE guidance include slimming substances, as well as skin lightening and darkening drugs.</p>
<p>Professor David Croisdale-Appleby, chair of the guideline committee, said: “Raising awareness about drug use and giving clear information about their potentially devastating effects, along with practical advice, will help contribute to the wider issues of preventing drug misuse.”</p>
